## 0. 目标（一句话）
在「口语复盘」输入屏加一个「🎤 录音」按钮：录一段英文 → 转成文字 → 自动填进 `#revText` 文本框 → 用户点「复盘」走现有闭环。90% 复用现有 JS 大脑，唯一新增 = 火山「大模型录音文件识别」转写。

## 1. 已验证的火山 ASR 接口事实（★不可改，照抄）
- **鉴权凭证**：App ID + Access Token，由**用户在设置屏自己填**，存 IndexedDB（key：`volc_appid`、`volc_token`）。**绝不硬编码进代码、绝不进仓库**（同 DeepSeek/MiniMax key 的存法）。
- **Resource-Id**：`volc.seedasr.auc`（★就是这个，不是 `bigasr`/`turbo`，那些用户没开通会报 45000030 not granted）。
- **异步两步**：submit（提交 base64 音频）→ 轮询 query（拿结果）。实测 ~20s 音频约 1–2s 出结果。
- **submit**：`POST https://openspeech.bytedance.com/api/v3/auc/bigmodel/submit`
  - headers：`X-Api-App-Key: <appid>`、`X-Api-Access-Key: <token>`、`X-Api-Resource-Id: volc.seedasr.auc`、`X-Api-Request-Id: <自己生成的uuid>`、`X-Api-Sequence: -1`、`Content-Type: application/json`
  - body：`{"user":{"uid":"situ"},"audio":{"format":"wav","data":"<base64音频，不含 data: 前缀>"},"request":{"model_name":"bigmodel","enable_itn":true,"enable_punc":true}}`
  - 成功：HTTP 200，body `{}`。
- **query**：`POST https://openspeech.bytedance.com/api/v3/auc/bigmodel/query`
  - headers：同上的 `X-Api-App-Key`/`X-Api-Access-Key`/`X-Api-Resource-Id`/`X-Api-Request-Id`（★request-id 必须与 submit 用**同一个值**，query 靠它找任务）、`Content-Type: application/json`。**不需要 X-Tt-Logid**（实测 reqid 单独就够；这点很关键，因为原生桥拿不到响应头）。
  - body：`{}`
- **★原生桥只回 `{status, body}`，拿不到响应头**（`X-Api-Status-Code` 读不到）。所以 JS 只能靠 **HTTP status + body 内容**判断状态。实测 body 形状：
  - submit 成功：HTTP 200，body `{}`
  - query 处理中：HTTP 200，body `{"audio_info":{},"result":{"text":""}}`（text 空、audio_info 无 duration）
  - query 完成：HTTP 200，body 含 `audio_info.duration`（数字）+ `result.text`（识别文本，可能空=纯静音）
  - 出错（如未开通/鉴权失败）：HTTP 4xx，body `{"header":{"code":45000030,"message":"..."}}`
  - **判定完成 = `body.audio_info && body.audio_info.duration` 存在**（比"text 非空"更稳，能兼容静音）；`result.text` 拿去用（空则提示重录）。

实现（照抄，逻辑已验证）：
```js
// ── 语音转写（阶段7：火山大模型录音文件识别） ──────────────────────────────
const _VOLC_SUBMIT = "https://openspeech.bytedance.com/api/v3/auc/bigmodel/submit";
const _VOLC_QUERY  = "https://openspeech.bytedance.com/api/v3/auc/bigmodel/query";
const _VOLC_RID    = "volc.seedasr.auc";

function _uuid() {
  if (typeof crypto !== "undefined" && crypto.randomUUID) return crypto.randomUUID();
  return "xxxxxxxx-xxxx-4xxx-yxxx-xxxxxxxxxxxx".replace(/[xy]/g, (c) => {
    const r = (Math.random() * 16) | 0, v = c === "x" ? r : (r & 0x3) | 0x8;
    return v.toString(16);
  });
}

/**
 * transcribe_audio — 把一段 base64 wav 音频转成文字
 * @param {{audioBase64:string, format?:string}} args  audioBase64 不含 "data:" 前缀
 * @returns {Promise<{ok:true, text:string}|{ok:false, error:string}>}
 */
async function transcribe_audio({ audioBase64, format = "wav" } = {}) {
  const b64 = (audioBase64 || "").trim();
  if (!b64) return { ok: false, error: "没有录到音频" };

  const appid = ((await settingsGet("volc_appid", "")) || "").trim();
  const token = ((await settingsGet("volc_token", "")) || "").trim();
  if (!appid || !token) {
    return { ok: false, error: "还没配置火山语音的 App ID / Access Token，请到「设置 → 语音转写」里填写" };
  }

  const reqid = _uuid();
  const headers = {
    "Content-Type": "application/json",
    "X-Api-App-Key": appid,
    "X-Api-Access-Key": token,
    "X-Api-Resource-Id": _VOLC_RID,
    "X-Api-Request-Id": reqid,
    "X-Api-Sequence": "-1",
  };

  // 1) submit
  let subResp;
  try {
    subResp = await httpPost(_VOLC_SUBMIT, headers, {
      user: { uid: "situ" },
      audio: { format, data: b64 },
      request: { model_name: "bigmodel", enable_itn: true, enable_punc: true },
    });
  } catch (e) {
    return { ok: false, error: "提交失败：" + ((e && e.message) || e) };
  }
  if (subResp.status !== 200) {
    let msg = "提交失败（HTTP " + subResp.status + "）";
    try { const j = await subResp.json(); if (j.header && j.header.message) msg = "转写服务：" + j.header.message; } catch (_) {}
    if (subResp.status === 403 || subResp.status === 401) msg += "（凭证或开通状态有误，请检查设置里的 App ID / Access Token）";
    return { ok: false, error: msg };
  }

  // 2) 轮询 query（headers 复用 submit 那套，request-id 同一个）
  const qHeaders = {
    "Content-Type": "application/json",
    "X-Api-App-Key": appid,
    "X-Api-Access-Key": token,
    "X-Api-Resource-Id": _VOLC_RID,
    "X-Api-Request-Id": reqid,
  };
  for (let i = 0; i < 30; i++) {
    await new Promise((r) => setTimeout(r, 1000));
    let qResp, j;
    try {
      qResp = await httpPost(_VOLC_QUERY, qHeaders, {});
      j = await qResp.json();
    } catch (e) {
      continue; // 单次网络抖动，继续轮询
    }
    // 错误 body
    if (j && j.header && j.header.code && j.header.code !== 0
        && j.header.code !== 20000001 && j.header.code !== 20000002) {
      return { ok: false, error: "转写失败：" + (j.header.message || ("code " + j.header.code)) };
    }
    if (qResp.status !== 200 && qResp.status !== 0) {
      return { ok: false, error: "转写失败（HTTP " + qResp.status + "）" };
    }
    // 完成判定：audio_info.duration 出现
    if (j && j.audio_info && j.audio_info.duration != null) {
      const text = (j.result && j.result.text) || "";
      return { ok: true, text };
    }
    // 否则处理中，继续
  }
  return { ok: false, error: "转写超时，请重试或缩短录音" };
}
```
在 `export const LocalApi = {...}` 的口语复盘那一行附近加：`transcribe_audio,`。

- 用 Web Audio 录（安卓 WebView MediaRecorder 输出 webm/opus，火山不认；必须自己出 wav）。参考编码器（照抄，标准做法）：
```js
let _recState = null; // {ctx, stream, proc, chunks, sr, timer, secs}
async function _toggleRecord() {
  const btn = $('#revRec'), lab = $('#revRecLab');
  if (_recState) { await _stopRecord(); return; }
  let stream;
  try {
    stream = await navigator.mediaDevices.getUserMedia({ audio: { channelCount: 1, echoCancellation: true, noiseSuppression: true } });
  } catch (e) { toast('需要麦克风权限：请在系统设置里允许四土使用麦克风'); return; }
  const ctx = new (window.AudioContext || window.webkitAudioContext)();
  const src = ctx.createMediaStreamSource(stream);
  const proc = ctx.createScriptProcessor(4096, 1, 1);
  const chunks = [];
  proc.onaudioprocess = (e) => { chunks.push(new Float32Array(e.inputBuffer.getChannelData(0))); };
  src.connect(proc); proc.connect(ctx.destination);
  _recState = { ctx, stream, proc, src, chunks, sr: ctx.sampleRate, secs: 0, timer: null };
  btn.classList.add('recording');
  const paint = () => {
    const m = Math.floor(_recState.secs / 60), s = String(_recState.secs % 60).padStart(2, '0');
    lab.innerHTML = `录音中 ${m}:${s} · 点击停止`;
    btn.innerHTML = _RS.stop + `<span id="revRecLab">${lab.innerHTML}</span>`;
  };
  paint();
  _recState.timer = setInterval(() => { _recState.secs++; paint(); if (_recState.secs >= 60) { toast('已到 60 秒上限'); _stopRecord(); } }, 1000);
}
async function _stopRecord() {
  if (!_recState) return;
  const st = _recState; _recState = null;
  clearInterval(st.timer);
  try { st.proc.disconnect(); st.src.disconnect(); } catch (_) {}
  try { st.stream.getTracks().forEach((t) => t.stop()); } catch (_) {}
  const btn = $('#revRec');
  btn.classList.remove('recording');
  // 拼接 + 降采样到 16k + 编 WAV
  let len = 0; st.chunks.forEach((c) => len += c.length);
  const flat = new Float32Array(len); let off = 0;
  st.chunks.forEach((c) => { flat.set(c, off); off += c.length; });
  const wavB64 = _encodeWavBase64(flat, st.sr, 16000);
  try { st.ctx.close(); } catch (_) {}
  if (!len) { _resetRecBtn(); toast('没录到声音，请重试'); return; }
  btn.disabled = true; btn.innerHTML = _RS.mic + '<span id="revRecLab">转写中…</span>';
  let res;
  try { res = await api('transcribe_audio', { audioBase64: wavB64, format: 'wav' }); }
  catch (e) { btn.disabled = false; _resetRecBtn(); toast('转写请求失败：' + e.message); return; }
  btn.disabled = false; _resetRecBtn();
  if (!res || !res.ok) { toast(res?.error || '转写失败'); return; }
  const t = (res.text || '').trim();
  if (!t) { toast('没听清，请靠近麦克风重录'); return; }
  const ta = $('#revText');
  ta.value = ta.value.trim() ? (ta.value.trim() + ' ' + t) : t;
  try { ta.dispatchEvent(new Event('input')); } catch (_) {}
  toast('已转写');
}
function _resetRecBtn() {
  const btn = $('#revRec'); if (!btn) return;
  btn.classList.remove('recording');
  btn.innerHTML = _RS.mic + '<span id="revRecLab">录音说一段</span>';
}
// Float32 → 目标采样率 → 16-bit PCM WAV → base64（不含 data: 前缀）
function _encodeWavBase64(samples, inRate, outRate) {
  let data = samples;
  if (outRate !== inRate) {
    const ratio = inRate / outRate;
    const outLen = Math.round(samples.length / ratio);
    data = new Float32Array(outLen);
    for (let i = 0; i < outLen; i++) {
      const idx = i * ratio, i0 = Math.floor(idx), i1 = Math.min(i0 + 1, samples.length - 1);
      const frac = idx - i0;
      data[i] = samples[i0] * (1 - frac) + samples[i1] * frac;
    }
  }
  const buf = new ArrayBuffer(44 + data.length * 2);
  const view = new DataView(buf);
  const ws = (o, s) => { for (let i = 0; i < s.length; i++) view.setUint8(o + i, s.charCodeAt(i)); };
  ws(0, 'RIFF'); view.setUint32(4, 36 + data.length * 2, true); ws(8, 'WAVE');
  ws(12, 'fmt '); view.setUint32(16, 16, true); view.setUint16(20, 1, true); view.setUint16(22, 1, true);
  view.setUint32(24, outRate, true); view.setUint32(28, outRate * 2, true); view.setUint16(32, 2, true); view.setUint16(34, 16, true);
  ws(36, 'data'); view.setUint32(40, data.length * 2, true);
  let o = 44;
  for (let i = 0; i < data.length; i++) { let s = Math.max(-1, Math.min(1, data[i])); view.setInt16(o, s < 0 ? s * 0x8000 : s * 0x7fff, true); o += 2; }
  // ArrayBuffer → base64（分块避免超大字符串栈溢出）
  const bytes = new Uint8Array(buf); let bin = '';
  const CH = 0x8000;
  for (let i = 0; i < bytes.length; i += CH) bin += String.fromCharCode.apply(null, bytes.subarray(i, i + CH));
  return btoa(bin);
}
```
